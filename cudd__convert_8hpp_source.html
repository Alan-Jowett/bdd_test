64inline std::pair<std::unique_ptr<Cudd>, BDD> convert_to_cudd_bdd(
65 const my_expression& expr, const std::unordered_set<std::string>& variable_names) {
66 // Create CUDD manager
67 auto cudd_mgr = std::make_unique<Cudd>();
68
69 // Create variable mapping
70 std::vector<std::string> sorted_vars(variable_names.begin(), variable_names.end());
71 std::ranges::sort(sorted_vars);
72
73 std::cout << "CUDD variable ordering: ";
74 for (size_t i = 0; i < sorted_vars.size(); ++i) {
75 std::cout << sorted_vars[i] << "=" << i;
76 if (i < sorted_vars.size() - 1)
77 std::cout << ", ";
78 }
79 std::cout << std::endl;
80
81 std::unordered_map<std::string, int> var_map;
82 for (size_t i = 0; i < sorted_vars.size(); ++i) {
83 var_map[sorted_vars[i]] = static_cast<int>(i);
84 }
85
86 // Recursive lambda to convert expression to CUDD BDD
87 std::function<BDD(const my_expression&)> convert_recursive =
88 [&](const my_expression& e) -> BDD {
89 return std::visit(
90 [&](const auto& content) -> BDD {
91 using T = std::decay_t<decltype(content)>;
92
93 if constexpr (std::is_same_v<T, my_variable>) {
94 auto it = var_map.find(content.variable_name);
95 if (it == var_map.end()) {
96 throw std::runtime_error("Variable not found in variable map: "
97 + content.variable_name);
98 }
99 return cudd_mgr->bddVar(it->second);
100 } else if constexpr (std::is_same_v<T, my_not>) {
101 BDD expr_bdd = convert_recursive(*content.expr);
102 BDD one = cudd_mgr->bddOne();
103 return expr_bdd ^ one;
104 } else if constexpr (std::is_same_v<T, my_and>) {
105 BDD left = convert_recursive(*content.left);
106 BDD right = convert_recursive(*content.right);
107 return left & right;
108 } else if constexpr (std::is_same_v<T, my_or>) {
109 BDD left = convert_recursive(*content.left);
110 BDD right = convert_recursive(*content.right);
111 return left | right;
112 } else if constexpr (std::is_same_v<T, my_xor>) {
113 BDD left = convert_recursive(*content.left);
114 BDD right = convert_recursive(*content.right);
115 return left ^ right;
116 }
117
118 // This should never be reached
119 throw std::runtime_error("Unknown expression type");
120 },
121 e);
122 };
123
124 // Convert the expression
125 BDD result = convert_recursive(expr);
126
127 return std::make_pair(std::move(cudd_mgr), result);
128}
