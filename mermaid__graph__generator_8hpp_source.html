214template <MermaidGraphIterator Iterator>
215void generate_mermaid_graph(const Iterator& root_iterator, std::ostream& out,
216 const MermaidConfig& config = MermaidConfig()) {
217 static_assert(MermaidGraphIterator<Iterator>);
218
219 if (config.show_frontmatter) {
220 out << "---\n";
221 out << "title: " << config.graph_title << "\n";
222 out << "---\n";
223 }
224
225 out << "flowchart " << config.direction << "\n";
226
227 graph_common::node_id_allocator id_alloc(config.node_id_prefix, config.node_id_start);
228 auto get_node_id = [&](const Iterator& iter) -> std::string {
229 return id_alloc.get_id(iter.get_node_address());
230 };
231
232 using graph_render_helpers::render_node_line;
233 auto build_node_definition = [&](const Iterator& iter, const std::string& node_id) {
234 std::string label = node_id;
235 if constexpr (has_get_label<Iterator>)
236 label = iter.get_label();
237 std::string shape = config.default_node_shape;
238 if constexpr (has_get_shape<Iterator>)
239 shape = iter.get_shape();
240 return render_node_line(node_id, label, shape);
241 };
242
243 using graph_render_helpers::render_edge_line;
244 auto build_edge_definition = [&](const Iterator& parent, const Iterator& child, size_t index,
245 const std::string& parent_id, const std::string& child_id) {
246 std::string style;
247 if constexpr (has_get_edge_style<Iterator>)
248 style = parent.get_edge_style(child, index);
249 std::string label;
250 if (config.show_edge_labels) {
251 if constexpr (has_get_edge_label<Iterator>)
252 label = parent.get_edge_label(child, index);
253 }
254 return render_edge_line(parent_id, child_id, style, label);
255 };
256
257 std::unordered_map<std::string, std::vector<std::string>> class_to_nodes;
258 std::vector<Iterator> unique_nodes;
259
260 // Collect nodes and edges in a single traversal to avoid double-walking
261 auto [unique_nodes_local, edges] =
262 dag_walker::collect_nodes_and_edges_topological(root_iterator);
263 unique_nodes = std::move(unique_nodes_local);
264
265 std::unordered_map<const void*, std::vector<dag_walker::EdgeInfo<Iterator>>> edges_by_parent;
266 for (const auto& edge : edges)
267 edges_by_parent[edge.parent.get_node_address()].push_back(edge);
268
269 std::vector<std::pair<std::string, const void*>> parent_ids_and_keys;
270 // Map node address -> emitted node id to avoid repeated linear searches
271 std::unordered_map<const void*, std::string> node_address_to_id;
272 // Collect node definition lines so we can sort them lexicographically
273 std::vector<std::string> node_lines;
274 node_lines.reserve(unique_nodes.size());
275 for (const auto& node : unique_nodes) {
276 std::string node_id = get_node_id(node);
277 node_address_to_id[node.get_node_address()] = node_id;
278 node_lines.push_back(build_node_definition(node, node_id));
279
280 if (config.show_css_classes) {
281 std::string css_class;
282 if constexpr (has_get_css_class<Iterator>) {
283 css_class = node.get_css_class();
284 } else if constexpr (has_get_label<Iterator>) {
285 auto it = config.label_to_class_map.find(node.get_label());
286 css_class =
287 (it != config.label_to_class_map.end()) ? it->second : config.default_css_class;
288 } else {
289 css_class = config.default_css_class;
290 }
291 if (!css_class.empty())
292 class_to_nodes[css_class].push_back(node_id);
293 }
294 }
295
296 // Sort node lines lexicographically and emit
297 std::sort(node_lines.begin(), node_lines.end());
298 for (const auto& line : node_lines)
299 out << line;
300
301 // collect parent ids in numeric order-aware list for topological emission
302 for (const auto& [parent_key, _] : edges_by_parent) {
303 auto it = node_address_to_id.find(parent_key);
304 if (it != node_address_to_id.end())
305 parent_ids_and_keys.push_back({it->second, parent_key});
306 }
307
308 std::sort(
309 parent_ids_and_keys.begin(), parent_ids_and_keys.end(),
310 [](const auto& a, const auto& b) { return graph_common::node_id_less(a.first, b.first); });
311 if (!edges.empty() && !unique_nodes.empty())
312 out << "\n";
313
314 // Collect all edge definition lines, sort lexicographically, then emit.
315 std::vector<std::string> edge_lines;
316 edge_lines.reserve(edges.size());
317 for (const auto& [parent_id, parent_key] : parent_ids_and_keys) {
318 auto& parent_edges = edges_by_parent[parent_key];
319 std::sort(parent_edges.begin(), parent_edges.end(),
320 [](const auto& a, const auto& b) { return a.child_index < b.child_index; });
321 for (const auto& edge : parent_edges) {
322 const void* parent_addr = edge.parent.get_node_address();
323 const void* child_addr = edge.child.get_node_address();
324 auto pit = node_address_to_id.find(parent_addr);
325 auto cit = node_address_to_id.find(child_addr);
326 const std::string& pid =
327 (pit != node_address_to_id.end()) ? pit->second : get_node_id(edge.parent);
328 const std::string& cid =
329 (cit != node_address_to_id.end()) ? cit->second : get_node_id(edge.child);
330 edge_lines.push_back(
331 build_edge_definition(edge.parent, edge.child, edge.child_index, pid, cid));
332 }
333 }
334
335 std::sort(edge_lines.begin(), edge_lines.end());
336 for (const auto& line : edge_lines)
337 out << line;
338
339 if (config.show_css_classes && !class_to_nodes.empty()) {
340 out << "\n";
341 auto sorted_class_assignments =
342 graph_render_helpers::flatten_and_sort_class_assignments(class_to_nodes);
343 for (const auto& [node_id, class_name] : sorted_class_assignments) {
344 out << " class " << node_id << " " << class_name << "\n";
345 }
346
347 if (!config.class_definitions.empty()) {
348 out << "\n";
349 for (const auto& [class_name, def_body] : config.class_definitions) {
350 out << " classDef " << class_name << " " << def_body << "\n";
351 }
352 }
353 }
354}
