25static bool validate_mermaid_flowchart(const string &s, std::string &err) {
26 // Basic checks: contains 'flowchart' and direction (TD or LR etc.)
27 if (s.find("flowchart") == string::npos) {
28 err = "Missing flowchart declaration";
29 return false;
30 }
31 if (s.find("TD") == string::npos && s.find("LR") == string::npos
32 && s.find("RL") == string::npos) {
33 err = "Missing flow direction";
34 return false;
35 }
36
37 // Check for balanced parentheses and brackets used in node shapes
38 int paren = 0, square = 0, brace = 0;
39 for (size_t i = 0; i < s.size(); ++i) {
40 char c = s[i];
41 if (c == '(')
42 ++paren;
43 if (c == ')')
44 --paren;
45 if (c == '[')
46 ++square;
47 if (c == ']')
48 --square;
49 if (c == '{')
50 ++brace;
51 if (c == '}')
52 --brace;
53 if (paren < 0 || square < 0 || brace < 0) {
54 err = "Unbalanced brackets";
55 return false;
56 }
57 }
58 if (paren != 0 || square != 0 || brace != 0) {
59 err = "Unbalanced brackets at end";
60 return false;
61 }
62
63 // Check presence of at least one arrow ('-->' or '-.->')
64 if (s.find("-->") == string::npos && s.find("-.->") == string::npos
65 && s.find("==>") == string::npos) {
66 err = "No edges detected";
67 return false;
68 }
69
70 return true;
71}

73static void validate_and_require(const std::string &s) {
74 std::string err;
75 bool ok = validate_mermaid_flowchart(s, err);
76 if (!ok)
77 FAIL("Mermaid validation failed: " + err);
78 REQUIRE(ok);
79}
