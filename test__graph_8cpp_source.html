29// Handle & EdgeRef:
30struct AdjHandle {
31 std::uint32_t id{};
32 std::uint64_t stable_key() const {
33 return id;
34 }
35 const void* debug_address() const {
36 return nullptr;
37 }
38 friend bool operator==(AdjHandle a, AdjHandle b) {
39 return a.id == b.id;
40 }
41 friend bool operator!=(AdjHandle a, AdjHandle b) {
42 return !(a == b);
43 }
44};

46struct AdjEdge {
47 AdjHandle tgt;
48 float w;
49 AdjHandle target() const {
50 return tgt;
51 }
52 float label() const {
53 return w;
54 }
55};

57struct AdjView {
58 using handle = AdjHandle;
59
60 const AdjDAG* g{};
61 auto children(handle h) const {
62 std::vector<AdjEdge> outv;
63 const auto& src = g->out[h.id];
64 outv.reserve(src.size());
65 for (const auto& e : src)
66 outv.push_back(AdjEdge{AdjHandle{e.to}, e.w});
67 return outv;
68 }
69 auto roots() const {
70 std::vector<AdjHandle> rv;
71 rv.reserve(g->roots_.size());
72 for (auto x : g->roots_)
73 rv.push_back(AdjHandle{x});
74 return rv;
75 }
76};

79// Test topo_order:
80TEST_CASE("topo_order on AdjView", "[graph][topo_order]") {
81 AdjDAG dag;
82 dag.out = {
83 {{1, 1.0f}, {2, 2.0f}}, // 0 -> 1,2
84 {{3, 1.0f}}, // 1 -> 3
85 {{3, 1.0f}}, // 2 -> 3
86 {} // 3 ->
87 };
88 dag.roots_ = {0};
89
90 AdjView view{&dag};
91 auto order = graph::topo_order(view, view.roots());
92
93 REQUIRE(order.size() == 4);
94 REQUIRE(order[0].id == 0);
95 REQUIRE((order[1].id == 1 || order[1].id == 2));
96 REQUIRE((order[2].id == 1 || order[2].id == 2));
97 REQUIRE(order[1].id != order[2].id);
98 REQUIRE(order[3].id == 3);
99}

103 AdjDAG dag;

105 {{1, 1.0f}, {2, 2.0f}}, // 0 -> 1,2

110 dag.roots_ = {0};
111 AdjView view{&dag};
112 auto sum_weights = graph::postorder_fold<AdjView, float>(
113 view, AdjHandle{0}, [](const AdjView& g, AdjHandle h, std::span<const float> child_sums) {
114 float total = 0.0f;
115 for (const auto& e : g.children(h)) {
116 total += e.label(); // edge weight
117 }
118 for (float cs : child_sums) {
119 total += cs; // sum from children
120 }
121 return total;
122 });
123 REQUIRE(sum_weights == Catch::Approx(5.0f)); // 1.0 + 2.0 + 1.0 + 1.0
124}
