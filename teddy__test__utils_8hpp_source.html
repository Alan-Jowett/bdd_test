12// Shared test helper for evaluating a TeDDy BDD with a boolean assignment
13inline bool evaluate_teddy_bdd(teddy::bdd_manager& manager, teddy::bdd_manager::diagram_t bdd,
14 const std::vector<bool>& assignment) {
15 using namespace teddy::ops;
16
17 // Guard against pathological assignment sizes overflowing `int`.
18 if (assignment.size() > static_cast<size_t>(INT_MAX)) {
19 throw std::out_of_range("Assignment size exceeds int range in test helper");
20 }
21
22 // NOTE: This helper assumes the provided `assignment` length does not exceed
23 // the number of variables configured in `manager`. Calling `manager.variable(i)`
24 // with i >= manager's variable count may lead to undefined behavior in TeDDy.
25 // Tests that need to exercise mismatched-length behavior should handle the
26 // guarded behavior explicitly (see tests/unit/test_teddy_graph.cpp).
27
28 teddy::bdd_manager::diagram_t cube = manager.constant(1);
29 for (size_t i = 0; i < assignment.size(); ++i) {
30 int idx = static_cast<int>(i);
31 teddy::bdd_manager::diagram_t var = manager.variable(idx);
32 if (!assignment[i]) {
33 teddy::bdd_manager::diagram_t one = manager.constant(1);
34 var = manager.apply<XOR>(var, one);
35 }
36 cube = manager.apply<AND>(cube, var);
37 }
38
39 teddy::bdd_manager::diagram_t result = manager.apply<AND>(bdd, cube);
40 return result.unsafe_get_root() == cube.unsafe_get_root();
41}
