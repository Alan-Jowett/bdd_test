31template <external_dag_view G>
32std::vector<typename G::handle> topo_order(const G& g, std::ranges::input_range auto roots) {
41 using H = typename G::handle;
42 static_assert(graph::node_handle<H>, "G::handle must model graph::node_handle");
43
44 std::unordered_map<std::uint64_t, std::size_t> indeg;
45 std::vector<H> order;
46
47 // 1) Build indegrees by a forward sweep from roots
48 std::vector<H> frontier(std::begin(roots), std::end(roots));
49 std::unordered_map<std::uint64_t, H> seen;
50
51 auto add = [&](H h) { seen.emplace(h.stable_key(), h); };
52 for (auto& r : frontier)
53 add(r);
54
55 // BFS to discover nodes and indegrees
56 for (std::size_t i = 0; i < frontier.size(); ++i) {
57 H u = frontier[i];
58 for (auto&& e : g.children(u)) {
59 H v = e.target();
60 ++indeg[v.stable_key()];
61 if (!seen.count(v.stable_key())) {
62 add(v);
63 frontier.push_back(v);
64 }
65 }
66 }
67
68 // 2) Kahnâ€™s algorithm
69 std::queue<H> q;
70 for (auto& [k, h] : seen) {
71 if (!indeg.count(k))
72 q.push(h); // indegree 0
73 }
74
75 while (!q.empty()) {
76 H u = q.front();
77 q.pop();
78 order.push_back(u);
79 for (auto&& e : g.children(u)) {
80 H v = e.target();
81 auto& d = indeg[v.stable_key()];
82 if (--d == 0)
83 q.push(v);
84 }
85 }
86
87 if (order.size() != seen.size())
88 throw std::logic_error("cycle detected (not a DAG)");
89
90 return order;
91}

110template <external_dag_view G, typename R, class Combine>
111R postorder_fold(const G& g, typename G::handle root, Combine&& combine) {
120 using H = typename G::handle;
121 static_assert(graph::node_handle<H>, "G::handle must model graph::node_handle");
122 std::unordered_map<std::uint64_t, R> memo;
123
124 std::function<R(H)> dfs = [&](H u) -> R {
125 auto k = u.stable_key();
126 if (auto it = memo.find(k); it != memo.end())
127 return it->second;
128
129 std::vector<R> child_vals;
130 for (auto&& e : g.children(u))
131 child_vals.push_back(dfs(e.target()));
132
133 R r = combine(g, u, std::span<const R>(child_vals));
134 memo.emplace(k, r);
135 return r;
136 };
137
138 return dfs(root);
139}
