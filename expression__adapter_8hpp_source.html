121 int32_t evaluate(int32_t left_val, int32_t right_val) const {
122 return std::visit(
123 [left_val, right_val](const auto& variant_expr) -> int32_t {
124 using T = std::decay_t<decltype(variant_expr)>;
125
126 if constexpr (std::is_same_v<T, my_and>) {
127 return (left_val != 0 && right_val != 0) ? 1 : 0;
128 } else if constexpr (std::is_same_v<T, my_or>) {
129 return (left_val != 0 || right_val != 0) ? 1 : 0;
130 } else if constexpr (std::is_same_v<T, my_xor>) {
131 return ((left_val != 0) != (right_val != 0)) ? 1 : 0;
132 } else if constexpr (std::is_same_v<T, my_not>) {
133 // For NOT, we only use the left value (right is ignored)
134 return (left_val != 0) ? 0 : 1;
135 } else {
136 throw std::runtime_error("evaluate() called on non-operation node");
137 }

172 void initialize_child_adapters() {
173 std::visit(
174 [this](const auto& variant_expr) {
175 using T = std::decay_t<decltype(variant_expr)>;
176
177 if constexpr (std::is_same_v<T, my_and> || std::is_same_v<T, my_or>
178 || std::is_same_v<T, my_xor>) {
179 if (variant_expr.left) {
180 left_adapter_ =
181 std::make_unique<expression_adapter>(*variant_expr.left, var_map_);
182 }
183 if (variant_expr.right) {
184 right_adapter_ =
185 std::make_unique<expression_adapter>(*variant_expr.right, var_map_);
186 }
187 } else if constexpr (std::is_same_v<T, my_not>) {
188 if (variant_expr.expr) {
189 left_adapter_ =
190 std::make_unique<expression_adapter>(*variant_expr.expr, var_map_);
191 // For NOT operations, TeDDy's from_expression_tree expects binary
192 // operations. Create a dummy right adapter with the same expression to
193 // satisfy this requirement.
194 right_adapter_ =
195 std::make_unique<expression_adapter>(*variant_expr.expr, var_map_);
196 }
