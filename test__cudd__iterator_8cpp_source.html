21TEST_CASE("cudd_iterator basic operations") {
22 // Create a CUDD manager and some simple BDDs
23 Cudd cudd_mgr;
24
25 std::vector<std::string> varnames = {"v0", "v1"};
26
27 BDD var0 = cudd_mgr.bddVar(0);
28 BDD var1 = cudd_mgr.bddVar(1);
29 BDD one = cudd_mgr.bddOne();
30 BDD zero = cudd_mgr.bddZero();
31
32 // Non-terminal variable node
33 cudd_iterator it_var0(cudd_mgr, var0.getNode(), &varnames);
34 REQUIRE(it_var0.is_valid());
35 REQUIRE(!it_var0.is_terminal());
36 REQUIRE(it_var0.get_label() == "v0");
37 REQUIRE(it_var0.get_variable_name() == "v0");
38 REQUIRE(it_var0.get_type() == "Variable");
39
40 auto children = it_var0.get_children();
41 REQUIRE(children.size() == 2);
42 REQUIRE(children[0].is_terminal());
43 REQUIRE(children[0].get_terminal_value() == 0);
44 REQUIRE(children[1].is_terminal());
45 REQUIRE(children[1].get_terminal_value() == 1);
46
47 // Complemented access to the same node should flip child terminal values
48 DdNode* not_var0_ptr = Cudd_Not(var0.getNode());
49 cudd_iterator it_not_var0(cudd_mgr, not_var0_ptr, &varnames);
50 auto children_not = it_not_var0.get_children();
51 REQUIRE(children_not.size() == 2);
52 REQUIRE(children_not[0].is_terminal());
53 REQUIRE(children_not[0].get_terminal_value() == 1);
54 REQUIRE(children_not[1].is_terminal());
55 REQUIRE(children_not[1].get_terminal_value() == 0);
56
57 // Node addresses should preserve complement bit and therefore differ
58 REQUIRE(it_var0.get_node_address() != it_not_var0.get_node_address());
59 REQUIRE(it_var0 != it_not_var0);
60 REQUIRE(it_var0 == it_var0);
61
62 // Terminal node attributes
63 cudd_iterator it_one(cudd_mgr, one.getNode(), nullptr);
64 REQUIRE(it_one.is_terminal());
65 REQUIRE(it_one.get_label() == "1");
66 auto attrs = it_one.get_dot_node_attributes();
67 REQUIRE(attrs.find("fillcolor=lightgreen") != std::string::npos);
68
69 // Edge attributes
70 REQUIRE(it_var0.get_dot_edge_attributes(children[0], 0).find("label=\"0\"")
71 != std::string::npos);
72 REQUIRE(it_var0.get_dot_edge_attributes(children[1], 1).find("label=\"1\"")
73 != std::string::npos);
74
75 // Tooltip and CSS class
76 REQUIRE(it_one.get_tooltip() == "1");
77 REQUIRE(it_one.get_css_class() == "terminal");
78 REQUIRE(it_var0.get_css_class() == "bddVariable");
79}

81TEST_CASE("cudd_iterator negative cases") {
82 Cudd cudd_mgr;
83
84 // Case: null varnames pointer for non-terminal variable
85 BDD var0 = cudd_mgr.bddVar(0);
86 DdNode* node0 = var0.getNode();
87
88 // Passing nullptr for varnames should still allow terminal checks,
89 // but get_label/get_variable_name should handle missing names.
90 cudd_iterator it_null_names(cudd_mgr, node0, nullptr);
91 REQUIRE(it_null_names.is_valid());
92 REQUIRE(!it_null_names.is_terminal());
93 // Label and variable name should not throw; they may return fallback values
94 REQUIRE_NOTHROW(it_null_names.get_label());
95 REQUIRE_NOTHROW(it_null_names.get_variable_name());
96
97 // Case: create a manager with fewer variables and use a node index that's out of range
98 Cudd small_mgr(1); // only 1 variable available
99 BDD v0 = small_mgr.bddVar(0);
100
101 // Create a fake node pointer by using a node from a different manager
102 DdNode* foreign_node = var0.getNode(); // from cudd_mgr, not small_mgr
103
104 // Constructing an iterator with a foreign node may be invalid; ensure code doesn't crash
105 REQUIRE_NOTHROW(cudd_iterator(small_mgr, foreign_node, nullptr));
106}
