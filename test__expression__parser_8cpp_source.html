36// Helper function to collect variables from an expression
37void collect_variables_helper(const my_expression& expr, std::unordered_set<std::string>& vars) {
38 std::visit(
39 [&vars](const auto& node) {
40 using T = std::decay_t<decltype(node)>;
41 if constexpr (std::is_same_v<T, my_variable>) {
42 vars.insert(node.variable_name);
43 } else if constexpr (std::is_same_v<T, my_and> || std::is_same_v<T, my_or>
44 || std::is_same_v<T, my_xor>) {
45 if (node.left) {
46 collect_variables_helper(*node.left, vars);
47 }
48 if (node.right) {
49 collect_variables_helper(*node.right, vars);
50 }
51 } else if constexpr (std::is_same_v<T, my_not>) {
52 if (node.expr) {
53 collect_variables_helper(*node.expr, vars);
54 }
55 }
56 },
57 expr);
58}
