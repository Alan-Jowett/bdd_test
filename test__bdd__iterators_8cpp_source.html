23void test_expression_to_cudd_and_teddy(const std::string& file_name) {
24 // Parse expression from file
25 auto expression = read_expression_from_file(file_name);
26 REQUIRE(expression != nullptr);
27
28 // Dynamically determine the number of variables needed
29 std::unordered_set<std::string> variable_names;
30 collect_variables_with_dag_walker(*expression, variable_names);
31
32 // Create sorted variable names for consistent ordering (same as in convert_to_bdd)
33 std::vector<std::string> sorted_variable_names(variable_names.begin(), variable_names.end());
34 std::ranges::sort(sorted_variable_names);
35
36 // Create a BDD manager with the appropriate number of variables
37 teddy::bdd_manager manager(static_cast<int>(variable_names.size()), 1'000);
38
39 auto teddy_bdd = convert_to_bdd(*expression, manager);
40 REQUIRE(teddy_bdd.unsafe_get_root() != nullptr);
41
42 auto [cudd_manager, cudd_bdd] = convert_to_cudd_bdd(*expression, variable_names);
43
44 teddy_iterator teddy_it(teddy_bdd.unsafe_get_root(), &sorted_variable_names);
45
46 // Collect unique nodes using dag_walker
47 auto [teddy_nodes, teddy_edges] = dag_walker::collect_nodes_and_edges_topological(teddy_it);
48 (void)teddy_edges;
49
50 REQUIRE(!teddy_nodes.empty());
51
52 // Get list of nodes using dag_walker for CUDD BDD in topological order
53 cudd_iterator cudd_it(*cudd_manager, cudd_bdd.getNode(), &sorted_variable_names);
54 auto [cudd_nodes, cudd_edges] = dag_walker::collect_nodes_and_edges_topological(cudd_it);
55 (void)cudd_edges;
56 REQUIRE(!cudd_nodes.empty());
57
58 REQUIRE(teddy_nodes.size() == cudd_nodes.size());
59
60 for (size_t i = 0; i < teddy_nodes.size(); ++i) {
61 REQUIRE(teddy_nodes[i].get_label() == cudd_nodes[i].get_label());
62 }
63}
