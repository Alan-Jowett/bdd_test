227template <DotGraphIterator Iterator>
228void generate_dot_graph(const Iterator& root_iterator, std::ostream& out,
229 const DotConfig& config = DotConfig()) {
230 static_assert(DotGraphIterator<Iterator>,
231 "Iterator must satisfy DotGraphIterator concept for DOT graph generation");
232
233 // Write DOT header with configuration
234 out << "digraph " << config.graph_name << " {\n";
235
236 if (!config.rankdir.empty()) {
237 out << " rankdir=" << config.rankdir << ";\n";
238 }
239
240 if (!config.font_name.empty()) {
241 out << " node [fontname=\"" << config.font_name << "\"];\n";
242 out << " edge [fontname=\"" << config.font_name << "\"];\n";
243 }
244
245 if (!config.default_node_shape.empty() || !config.default_node_style.empty()) {
246 out << " node [";
247 bool first = true;
248 if (!config.default_node_shape.empty()) {
249 out << "shape=" << config.default_node_shape;
250 first = false;
251 }
252 if (!config.default_node_style.empty()) {
253 if (!first)
254 out << ", ";
255 out << "style=" << config.default_node_style;
256 }
257 out << "];\n";
258 }
259
260 if (!config.default_edge_style.empty()) {
261 out << " edge [style=" << config.default_edge_style << "];\n";
262 }
263
264 out << "\n";
265
266 // Node ID management for unique DOT identifiers
267 // Use shared node_id_allocator to produce stable node identifiers
268 graph_common::node_id_allocator id_alloc("node");
269
270 auto get_node_id = [&](const Iterator& iter) -> std::string {
271 const void* key = iter.get_node_address();
272 return id_alloc.get_id(key);
273 };
274
275 // Generate node attributes from iterator properties using shared helpers
276 auto build_node_attributes = [&](const Iterator& iter,
277 const std::string& node_id) -> std::string {
278 std::vector<std::string> props;
279
280 // Label (use iterator label or fallback to node_id)
281 if constexpr (has_get_label<Iterator>) {
282 props.push_back("label = \"" + graph_render_helpers::escape_label(iter.get_label())
283 + "\"");
284 } else {
285 props.push_back("label = \"" + graph_render_helpers::escape_label(node_id) + "\"");
286 }
287
288 // Optional node properties
289 if constexpr (has_get_shape<Iterator>) {
290 props.push_back("shape=" + iter.get_shape());
291 }
292 if constexpr (has_get_style<Iterator>) {
293 props.push_back("style = " + iter.get_style());
294 }
295 if constexpr (has_get_fillcolor<Iterator>) {
296 props.push_back("fillcolor = " + iter.get_fillcolor());
297 }
298 if constexpr (has_get_fontcolor<Iterator>) {
299 props.push_back("fontcolor = " + iter.get_fontcolor());
300 }
301 if constexpr (has_get_tooltip<Iterator>) {
302 props.push_back("tooltip = \"" + graph_render_helpers::escape_label(iter.get_tooltip())
303 + "\"");
304 }
305
306 return graph_render_helpers::join_dot_properties(props);
307 };
308
309 // Generate edge attributes from iterator properties
310 auto build_edge_attributes = [](const Iterator& parent, const Iterator& child,
311 size_t index) -> std::string {
312 std::vector<std::string> properties;
313
314 if constexpr (has_get_edge_label<Iterator>) {
315 std::string label = parent.get_edge_label(child, index);
316 if (!label.empty()) {
317 properties.push_back("label = \"" + graph_render_helpers::escape_label(label)
318 + "\"");
319 }
320 }
321 if constexpr (has_get_edge_style<Iterator>) {
322 properties.push_back("style = " + parent.get_edge_style(child, index));
323 }
324 if constexpr (has_get_edge_color<Iterator>) {
325 properties.push_back("color = " + parent.get_edge_color(child, index));
326 }
327 if constexpr (has_get_edge_fontcolor<Iterator>) {
328 properties.push_back("fontcolor = " + parent.get_edge_fontcolor(child, index));
329 }
330
331 return graph_render_helpers::join_dot_properties(properties);
332 };
333
334 // Collect nodes and edges in a single traversal to avoid double-walking
335 auto [unique_nodes, edges] = dag_walker::collect_nodes_and_edges_topological(root_iterator);
336
337 if (config.use_bdd_format) {
338 // BDD-specific grouped shape format
339
340 // First, create node ID mapping by touching each node
341 for (const auto& node : unique_nodes) {
342 id_alloc.get_id(node.get_node_address());
343 }
344
345 // Group nodes by shape for BDD-style declarations
346 std::vector<std::string> square_nodes;
347 std::vector<std::string> circle_nodes;
348
349 for (const auto& node : unique_nodes) {
350 std::string node_id = id_alloc.get_id(node.get_node_address());
351 std::string shape = "circle"; // default
352 if constexpr (has_get_shape<Iterator>) {
353 shape = node.get_shape();
354 }
355
356 if (shape == "square") {
357 square_nodes.push_back(node_id);
358 } else {
359 circle_nodes.push_back(node_id); // Default to circle
360 }
361 }
362
363 // Output shape declarations (BDD-specific format)
364 if (!square_nodes.empty()) {
365 out << " node [shape = square]";
366 for (const auto& node_id : square_nodes) {
367 out << " " << node_id;
368 }
369 out << ";\n";
370 out << " node [shape = circle];\n\n";
371 }
372
373 // Output individual node declarations (without shape attributes for BDD format)
374 std::vector<std::string> node_lines;
375 node_lines.reserve(unique_nodes.size());
376 for (const auto& node : unique_nodes) {
377 std::string node_id = id_alloc.get_id(node.get_node_address());
378 std::vector<std::string> props;
379 if constexpr (has_get_label<Iterator>) {
380 props.push_back("label = \"" + graph_render_helpers::escape_label(node.get_label())
381 + "\"");
382 } else {
383 props.push_back("label = \"" + graph_render_helpers::escape_label(node_id) + "\"");
384 }
385 if constexpr (has_get_tooltip<Iterator>) {
386 props.push_back("tooltip = \""
387 + graph_render_helpers::escape_label(node.get_tooltip()) + "\"");
388 }
389 node_lines.push_back(" " + node_id + " "
390 + graph_render_helpers::join_dot_properties(props) + ";\n");
391 }
392
393 std::sort(node_lines.begin(), node_lines.end());
394 for (const auto& line : node_lines) {
395 out << line;
396 }
397 } else {
398 // Standard format: individual node declarations with all attributes
399 std::vector<std::string> node_lines;
400 node_lines.reserve(unique_nodes.size());
401 for (const auto& node : unique_nodes) {
402 std::string node_id = get_node_id(node);
403 std::string node_attrs = build_node_attributes(node, node_id);
404 node_lines.push_back(" " + node_id + " " + node_attrs + ";\n");
405 }
406
407 std::sort(node_lines.begin(), node_lines.end());
408 for (const auto& line : node_lines) {
409 out << line;
410 }
411 }
412
413 // Use the edges collected above
414 out << "\n";
415 std::vector<std::string> edge_lines;
416 edge_lines.reserve(edges.size());
417 for (const auto& edge : edges) {
418 std::string parent_id = get_node_id(edge.parent);
419 std::string child_id = get_node_id(edge.child);
420 std::string edge_attrs = build_edge_attributes(edge.parent, edge.child, edge.child_index);
421
422 std::string line = " " + parent_id + " -> " + child_id;
423 if (!edge_attrs.empty()) {
424 line += " " + edge_attrs;
425 }
426 line += ";\n";
427 edge_lines.push_back(std::move(line));
428 }
429
430 std::sort(edge_lines.begin(), edge_lines.end());
431 for (const auto& line : edge_lines) {
432 out << line;
433 }
434
435 out << "}\n";
436}
