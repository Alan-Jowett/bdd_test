48template <typename T>
49concept NodeTableIterator = BaseGraphIterator<T> && requires(T t) {
50 // Additional requirements for table generation
51 { t.is_valid() } -> std::convertible_to<bool>;
52 { t.is_terminal() } -> std::convertible_to<bool>;
53 { t.get_variable_name() } -> std::convertible_to<std::string>;
54 { t.get_type() } -> std::convertible_to<std::string>;
55
56 // Terminal value (for terminal nodes)
57 { t.get_terminal_value() } -> std::convertible_to<int>;
58};

120template <NodeTableIterator Iterator>
121void generate_text_table(const Iterator& root_iterator, std::ostream& out,
122 const TextTableConfig& config = TextTableConfig()) {
123 static_assert(NodeTableIterator<Iterator>,
124 "Iterator must satisfy NodeTableIterator concept for text table generation");
125
126 // Collect all nodes (and edges, unused here) using DAG walker
127 auto [nodes, edges] = dag_walker::collect_nodes_and_edges_topological(root_iterator);
128 (void)edges;
129
130 // Reverse to achieve parents-before-children ordering with terminals at end
131 // (matching legacy BDD function behavior)
132 std::reverse(nodes.begin(), nodes.end());
133
134 // Create index mapping for child references
135 std::unordered_map<const void*, int> node_to_index;
136 for (size_t i = 0; i < nodes.size(); ++i) {
137 node_to_index[nodes[i].get_node_address()] = static_cast<int>(i);
138 }
139
140 // Write headers
141 if (config.include_headers) {
142 out << config.table_title << ":\n";
143 }
144 out << "Index" << config.separator << "Variable" << config.separator << "False Child"
145 << config.separator << "True Child" << config.separator << "Type\n";
146 out << config.header_separator << "\n";
147
148 // Write table rows
149 for (size_t i = 0; i < nodes.size(); ++i) {
150 const auto& iter = nodes[i];
151
152 out << std::format("{:>{}}", static_cast<int>(i), config.index_width) << config.separator;
153 out << std::format("{:>{}}", iter.get_variable_name(), config.variable_width)
154 << config.separator;
155
156 if (iter.is_terminal()) {
157 out << std::format("{:>{}}", "-", config.false_child_width) << config.separator;
158 out << std::format("{:>{}}", "-", config.true_child_width) << config.separator;
159 } else {
160 // Find child indices
161 auto children = iter.get_children();
162 int false_child_idx = -1;
163 int true_child_idx = -1;
164
165 if (children.size() >= 2) {
166 auto false_addr = children[0].get_node_address();
167 auto true_addr = children[1].get_node_address();
168
169 auto false_it = node_to_index.find(false_addr);
170 auto true_it = node_to_index.find(true_addr);
171
172 if (false_it != node_to_index.end())
173 false_child_idx = false_it->second;
174 if (true_it != node_to_index.end())
175 true_child_idx = true_it->second;
176 }
177
178 out << std::format("{:>{}}", false_child_idx, config.false_child_width)
179 << config.separator;
180 out << std::format("{:>{}}", true_child_idx, config.true_child_width)
181 << config.separator;
182 }
183 out << iter.get_type() << "\n";
184 }
185
186 if (config.include_headers) {
187 out << "\nTotal nodes: " << nodes.size() << "\n";
188 out << "Note: Topological order ensures parents appear before children.\n";
189 }
190}

367template <NodeTableIterator Iterator>
368void generate_table(const Iterator& root_iterator, std::ostream& out,
369 const std::string& format = "text", bool include_headers = true) {
370 static_assert(NodeTableIterator<Iterator>,
371 "Iterator must satisfy NodeTableIterator concept for table generation");
372
373 if (format == "text") {
374 generate_text_table(root_iterator, out, TextTableConfig(include_headers));
375 } else if (format == "markdown") {
376 generate_markdown_table(root_iterator, out, MarkdownTableConfig());
377 } else if (format == "csv") {
378 generate_csv_table(root_iterator, out, include_headers);
379 } else {
380 throw std::invalid_argument("Unsupported table format: " + format);
381 }
382}
