24template <class H>
25concept node_handle = std::copyable<H> && requires(H h) {
26 { h.stable_key() } -> std::convertible_to<std::uint64_t>; // memo key
27 { h == h } -> std::same_as<bool>;
28 { h != h } -> std::same_as<bool>;
29};

54template <class E, class H>
55concept edge_ref = requires(const E& e) {
56 { e.target() } -> std::convertible_to<H>;
57 // Optional: edge label/weight (ignored by algorithms that don't need it)
58 // { e.label() } -> /* any type */;

69template <class R, class H>
70concept children_range = std::ranges::input_range<R> && requires(R r) {
71 requires edge_ref<std::ranges::range_value_t<R>, H>
72 || edge_ref<std::remove_cvref_t<std::ranges::range_reference_t<R>>, H>;
73};

87template <class G>
88concept external_dag_view = requires(const G& g, typename G::handle h) {
89 typename G::handle;
90 requires node_handle<typename G::handle>;
91 // Children of a node as a range of edges:
92 requires std::ranges::input_range<decltype(g.children(h))>;
93 requires children_range<decltype(g.children(h)), typename G::handle>;
94 // Roots are optional; many algorithms accept an explicit root list
95 requires std::ranges::input_range<decltype(g.roots())>; // optional but nice to have
96};
