8// Minimal iterator that only satisfies BaseGraphIterator (no optional properties)
9struct minimal_node {

11 minimal_node() = default;
12 explicit minimal_node(std::initializer_list<minimal_node> ch) : children(ch) {}
13 std::vector<minimal_node> get_children() const {
14 return children;
15 }
16 const void* get_node_address() const {
17 return reinterpret_cast<const void*>(this);
18 }
19 bool operator==(const minimal_node& o) const {
20 return this == &o;
21 }
22 bool operator!=(const minimal_node& o) const {
23 return !(*this == o);
24 }
25};
