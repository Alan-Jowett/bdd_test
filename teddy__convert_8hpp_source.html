77teddy::bdd_manager::diagram_t inline convert_to_bdd(const my_expression& expr,
78 teddy::bdd_manager& mgr) {
79 using bdd_t = teddy::bdd_manager::diagram_t;
80 using namespace teddy::ops;
81
82 // First pass: collect all unique variable names using dag_walker
83 std::unordered_set<std::string> variable_names;
84 collect_variables_with_dag_walker(expr, variable_names);
85
86 // Build variable map with sorted variable names for consistent ordering
87 std::vector<std::string> sorted_vars(variable_names.begin(), variable_names.end());
88 std::ranges::sort(sorted_vars);
89
90 std::cout << "TeDDy variable ordering: ";
91 for (size_t i = 0; i < sorted_vars.size(); ++i) {
92 std::cout << sorted_vars[i] << "=" << i;
93 if (i < sorted_vars.size() - 1)
94 std::cout << ", ";
95 }
96 std::cout << std::endl;
97
98 std::unordered_map<std::string, int> var_map;
99 for (size_t i = 0; i < sorted_vars.size(); ++i) {
100 var_map[sorted_vars[i]] = static_cast<int>(i);
101 }
102
103 // Helper function for recursive conversion (no memoization needed for tree structure)
104 std::function<bdd_t(const my_expression&)> convert_recursive =
105 [&](const my_expression& e) -> bdd_t {
106 return std::visit(
107 [&](const auto& variant_expr) -> bdd_t {
108 using T = std::decay_t<decltype(variant_expr)>;
109
110 if constexpr (std::is_same_v<T, my_variable>) {
111 auto it = var_map.find(variant_expr.variable_name);
112 if (it == var_map.end()) {
113 throw std::runtime_error("Variable not found: "
114 + variant_expr.variable_name);
115 }
116 return mgr.variable(it->second);
117 } else if constexpr (std::is_same_v<T, my_and>) {
118 bdd_t left_bdd = convert_recursive(*variant_expr.left);
119 bdd_t right_bdd = convert_recursive(*variant_expr.right);
120 return mgr.apply<AND>(left_bdd, right_bdd);
121 } else if constexpr (std::is_same_v<T, my_or>) {
122 bdd_t left_bdd = convert_recursive(*variant_expr.left);
123 bdd_t right_bdd = convert_recursive(*variant_expr.right);
124 return mgr.apply<OR>(left_bdd, right_bdd);
125 } else if constexpr (std::is_same_v<T, my_not>) {
126 bdd_t expr_bdd = convert_recursive(*variant_expr.expr);
127 bdd_t one = mgr.constant(1);
128 return mgr.apply<XOR>(expr_bdd, one);
129 } else if constexpr (std::is_same_v<T, my_xor>) {
130 bdd_t left_bdd = convert_recursive(*variant_expr.left);
131 bdd_t right_bdd = convert_recursive(*variant_expr.right);
132 return mgr.apply<XOR>(left_bdd, right_bdd);
133 }
134 },
135 e);
136 };
137
138 return convert_recursive(expr);
139}
