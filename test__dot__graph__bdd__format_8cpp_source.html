30struct bdd_iter {
31 std::shared_ptr<BddNode> node;
32 bdd_iter() = default;
33 explicit bdd_iter(std::shared_ptr<BddNode> n) : node(std::move(n)) {}

35 std::vector<bdd_iter> get_children() const {
36 std::vector<bdd_iter> out;
37 if (!node)
38 return out;
39 std::transform(node->children.begin(), node->children.end(), std::back_inserter(out),
40 [](const std::shared_ptr<BddNode>& c) { return bdd_iter(c); });
41 return out;
42 }
43
44 const void* get_node_address() const {
45 return node ? reinterpret_cast<const void*>(node.get()) : nullptr;
46 }
47
48 bool operator==(const bdd_iter& other) const {
49 return get_node_address() == other.get_node_address();

51 bool operator!=(const bdd_iter& other) const {

90TEST_CASE("DotGenerator - BDD format and optional properties", "[dot_graph][bdd_format]") {
91 // Build a small DAG: root -> s1, root -> s2, s1 -> s2 (shared child)
92 auto s2 = std::make_shared<BddNode>("shared_node", "circle", "shared tip");
93 auto s1 = std::make_shared<BddNode>("square_node", "square", "square tip");
94 auto root = std::make_shared<BddNode>("root_node", "circle", "root tip");
95
96 s1->children = {s2};
97 root->children = {s1, s2};
98
99 bdd_iter root_iter(root);
100
101 std::ostringstream out;
102 dot_graph::DotConfig cfg;
103 cfg.graph_name = "BDDTest";
104 cfg.use_bdd_format = true; // exercise BDD grouping branch
105
106 REQUIRE_NOTHROW(dot_graph::generate_dot_graph(root_iter, out, cfg));
107
108 std::string dot = out.str();
109
110 // Basic structure
111 REQUIRE_THAT(dot, ContainsSubstring("digraph BDDTest"));
112 REQUIRE_THAT(dot, ContainsSubstring("shared_node"));
113 REQUIRE_THAT(dot, ContainsSubstring("square_node"));
114 REQUIRE_THAT(dot, ContainsSubstring("root_node"));
115
116 // BDD grouping should emit shape declaration for squares
117 REQUIRE_THAT(dot, ContainsSubstring("shape = square"));
118
119 // Tooltip content from iterator should appear
120 REQUIRE_THAT(dot, ContainsSubstring("shared tip"));
121
122 // Edge attributes emitted (style/color/label)
123 REQUIRE_THAT(dot, ContainsSubstring("dashed"));
124 REQUIRE_THAT(dot, ContainsSubstring("blue"));
125 REQUIRE_THAT(dot, ContainsSubstring("label = \"L\""));
126
127 // Edges exist
128 REQUIRE_THAT(dot, ContainsSubstring("->"));
129}
