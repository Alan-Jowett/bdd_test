70 std::vector<cudd_iterator> get_children() const {
71 std::vector<cudd_iterator> children;
72 children.reserve(2);
73
74 // Get the regular (non-complemented) physical node
75 DdNode* regular_node = Cudd_Regular(node_);
76
77 // Only non-constant nodes have children
78 if (!Cudd_IsConstant(regular_node)) {
79 // Get children from the regular node
80 DdNode* else_child = Cudd_E(regular_node);
81 DdNode* then_child = Cudd_T(regular_node);
82
83 // If this node is accessed through a complement edge, negate the children
84 // This implements NOT(f) = f with negated children, expanding complement edges
85 if (Cudd_IsComplement(node_)) {
86 else_child = Cudd_Not(else_child);
87 then_child = Cudd_Not(then_child);
88 }
89
90 // Return children in TeDDy convention order: FALSE (else) first, TRUE (then) second
91 children.emplace_back(*cudd_manager_, else_child, variable_names_);
92 children.emplace_back(*cudd_manager_, then_child, variable_names_);
93 }
94
95 return children;
96 }
