101template <DagWalkerIterator Iterator, Visitor<Iterator> V>
102void walk_dag_topological_order(const Iterator& root_iterator, V&& visitor) {
103 static_assert(DagWalkerIterator<Iterator>,
104 "Iterator must satisfy DagWalkerIterator concept for topological traversal");
105
106 std::unordered_set<const void*> visited_nodes;
107 std::unordered_set<const void*> completed_nodes;
108
109 // Helper to check if iterator has should_process method
110 auto should_process_impl = [&](const Iterator& iter) -> bool {
111 if constexpr (has_should_process<Iterator>) {
112 return iter.should_process();
113 } else {
114 return true; // Default behavior if method not provided
115 }
116 };
117
118 std::function<void(const Iterator&, size_t, const Iterator*)> walk_impl =
119 [&](const Iterator& current, size_t index_from_parent, const Iterator* parent) {
120 const void* node_key = current.get_node_address();
121
122 // Skip nodes that shouldn't be processed
123 if (!should_process_impl(current)) {
124 return;
125 }
126
127 // If we've already completed this node, don't process it again
128 if (completed_nodes.find(node_key) != completed_nodes.end()) {
129 return;
130 }
131
132 // Check if we're already in the process of visiting this node (cycle detection)
133 if (visited_nodes.find(node_key) != visited_nodes.end()) {
134 // We're in a cycle - treat as a revisit but don't recurse
135 NodeInfo<Iterator> node_info(current, index_from_parent, parent);
136 visitor(node_info);
137 return;
138 }
139 visited_nodes.insert(node_key);
140
141 // Process all children first (post-order traversal)
142 std::vector<Iterator> children = current.get_children();
143 for (size_t i = 0; i < children.size(); ++i) {
144 const Iterator& child = children[i];
145 walk_impl(child, i, &current);
146 }
147
148 // Now visit this node after all children have been processed
149 NodeInfo<Iterator> node_info(current, index_from_parent, parent);
150 visitor(node_info);
151
152 // Mark this node as completed
153 completed_nodes.insert(node_key);
154 };
155
156 // Start traversal from root
157 walk_impl(root_iterator, 0, nullptr);
158}

190template <DagWalkerIterator Iterator>
191std::pair<std::vector<Iterator>, std::vector<EdgeInfo<Iterator>>>
192collect_nodes_and_edges_topological(const Iterator& root_iterator) {
193 static_assert(DagWalkerIterator<Iterator>,
194 "Iterator must satisfy DagWalkerIterator concept for topological collection");
195
196 std::vector<Iterator> nodes;
197 std::vector<EdgeInfo<Iterator>> edges;
198
199 // Use the existing walk_dag_topological_order implementation to ensure
200 // cycle handling and should_process behavior remain consistent.
201 walk_dag_topological_order(root_iterator, [&](const NodeInfo<Iterator>& info) {
202 // Record the node in topological order
203 nodes.push_back(info.node);
204
205 // Record edges observed for this parent
206 auto children = info.node.get_children();
207 for (size_t i = 0; i < children.size(); ++i) {
208 edges.emplace_back(info.node, children[i], i);
209 }
210 });
211
212 return {nodes, edges};
213}
