58TEST_CASE("node_table: text/markdown/csv generation", "[node_table]") {
59 // Build a small graph: root -> (false: leaf0, true: leaf1)
60 FakeIterator leaf0;
61 leaf0.var = "leaf0";
62 leaf0.type = "terminal";
63 leaf0.term_value = 0;
64 FakeIterator leaf1;
65 leaf1.var = "leaf1";
66 leaf1.type = "terminal";
67 leaf1.term_value = 1;
68
69 FakeIterator root;
70 root.var = "x";
71 root.type = "internal";
72 root.children = {leaf0, leaf1};
73
74 std::ostringstream out;
75
76 SECTION("text table includes headers and counts") {
77 node_table::generate_text_table(root, out, node_table::TextTableConfig(true, "TestTitle"));
78 std::string s = out.str();
79 REQUIRE(s.find("TestTitle:") != std::string::npos);
80 REQUIRE(s.find("Index") != std::string::npos);
81 REQUIRE(s.find("Total nodes") != std::string::npos);
82 }
83
84 out.str("");
85 out.clear();
86 SECTION("markdown table format") {
87 node_table::generate_markdown_table(root, out, node_table::MarkdownTableConfig("left"));
88 std::string s = out.str();
89 REQUIRE(s.find("| Index |") != std::string::npos);
90 // Terminals are rendered as "- | - | - | <type> |" in markdown
91 REQUIRE(s.find("- | - | - | terminal |") != std::string::npos);
92 REQUIRE(s.find("internal") != std::string::npos);
93 }
94
95 out.str("");
96 out.clear();
97 SECTION("csv output") {
98 node_table::generate_csv_table(root, out, true);
99 std::string s = out.str();
100 REQUIRE(s.find("Index,Variable,False Child,True Child,Type") != std::string::npos);
101 REQUIRE(s.find("leaf1") != std::string::npos);
