30class ExpressionToken:
31 """Represents a token in a boolean expression.
32
33 A token is the smallest unit of meaning in an expression,
34 such as variables, operators, parentheses, or the end-of-file marker.
35
36 Attributes:
37 type (str): Token type ('variable', 'operator', 'lparen', 'rparen', 'eof')
38 value (str): The actual text value of the token
39 pos (int): Position in the source text where this token was found
40
41 Args:
42 token_type (str): The type classification of this token
43 value (str): The string content of the token
44 pos (int, optional): Source position. Defaults to 0.
45 """
46
47 def __init__(self, token_type: str, value: str, pos: int = 0):
48 self.type = token_type # 'variable', 'operator', 'lparen', 'rparen', 'eof'
49 self.value = value
50 self.pos = pos

53class ExpressionLexer:
54 """Tokenizes boolean expressions into structured tokens.
55
56 This lexer breaks down boolean expressions into individual tokens
57 that can be consumed by the parser. It handles variables, operators,
58 parentheses, and whitespace according to standard boolean expression syntax.
59
60 Supported operators:
61 - AND: '&', '&&', 'AND', 'and'
62 - OR: '|', '||', 'OR', 'or'
63 - NOT: '!', '~', 'NOT', 'not'
64 - XOR: '^', 'XOR', 'xor'
65
66 Attributes:
67 text (str): The input expression text
68 pos (int): Current position in the text
69 tokens (List[ExpressionToken]): Generated tokens
70
71 Args:
72 text (str): Boolean expression to tokenize
73 """
74
75 def __init__(self, text: str):
76 self.text = text
77 self.pos = 0
78 self.tokens = []
79 self._tokenize()
80
81 def _skip_whitespace(self):
82 """Skip whitespace characters"""
83 while self.pos < len(self.text) and self.text[self.pos].isspace():
84 self.pos += 1
85
86 def _read_word(self) -> str:
87 """Read a word (variable name or operator)"""
88 start = self.pos
89 while (self.pos < len(self.text) and
90 (self.text[self.pos].isalnum() or self.text[self.pos] in '_$@#!')):
91 self.pos += 1
92 return self.text[start:self.pos]
93
94 def _tokenize(self):
95 """Convert input text into tokens"""
96 while self.pos < len(self.text):
97 self._skip_whitespace()
98
99 if self.pos >= len(self.text):
100 break
101
102 char = self.text[self.pos]
103
104 if char == '(':
105 self.tokens.append(ExpressionToken('lparen', char, self.pos))
106 self.pos += 1
107 elif char == ')':
108 self.tokens.append(ExpressionToken('rparen', char, self.pos))
109 self.pos += 1
110 elif char.isalpha() or char in '_$@#!':
111 word = self._read_word()
112 if word.upper() in ['AND', 'OR', 'XOR', 'NOT']:
113 self.tokens.append(ExpressionToken('operator', word.upper(), self.pos - len(word)))
114 else:
115 self.tokens.append(ExpressionToken('variable', word, self.pos - len(word)))
116 else:
117 raise ValueError(f"Unexpected character '{char}' at position {self.pos}")
118
119 self.tokens.append(ExpressionToken('eof', '', self.pos))

122class ExpressionParser:
123 """Parses boolean expressions into BDD-compatible format.
124
125 This parser converts tokenized boolean expressions into a format
126 suitable for BDD construction. It follows standard operator precedence:
127 1. NOT (highest)
128 2. AND
129 3. OR, XOR (lowest, left-associative)
130
131 The parser validates syntax and tracks all variables used in the expression
132 for consistent variable ordering in BDD construction.
133
134 Attributes:
135 tokens (List[ExpressionToken]): Input tokens from lexer
136 pos (int): Current position in token stream
137 variables (Set[str]): Set of all variables found in expression
138
139 Args:
140 tokens (List[ExpressionToken]): Tokenized expression from ExpressionLexer
141
142 Raises:
143 ValueError: If expression has invalid syntax or unexpected tokens
144 """
145
146 def __init__(self, tokens: List[ExpressionToken]):
147 self.tokens = tokens
148 self.pos = 0
149 self.variables: Set[str] = set()
150
151 def current_token(self) -> ExpressionToken:
152 """Get the current token without advancing the position.
153
154 Returns:
155 ExpressionToken: The current token, or EOF token if at end of input.
156 Returns the last token (EOF) if position exceeds token count.
157 """
158 return self.tokens[self.pos] if self.pos < len(self.tokens) else self.tokens[-1]
159
160 def consume_token(self, expected_type: Optional[str] = None) -> ExpressionToken:
161 """Consume and return the current token, optionally validating its type.
162
163 Args:
164 expected_type (Optional[str]): Expected token type to validate against.
165 If provided and doesn't match current token type, raises ValueError.
166
167 Returns:
168 ExpressionToken: The consumed token before advancing position.
169
170 Raises:
171 ValueError: If expected_type is specified and doesn't match current token type.
172 """
173 token = self.current_token()
174 if expected_type and token.type != expected_type:
175 raise ValueError(f"Expected {expected_type} but got {token.type}")
176 self.pos += 1
177 return token
178
179 def parse_expression(self) -> str:
180 """Parse the complete boolean expression from tokens.
181
182 Parses the entire expression and validates that all tokens are consumed,
183 ensuring no unexpected tokens remain after parsing completes.
184
185 Returns:
186 str: The parsed expression string in prefix notation suitable for BDD construction.
187
188 Raises:
189 ValueError: If unexpected tokens remain after parsing the complete expression.
190 """
191 expr = self.parse_or_expression()
192 if self.current_token().type != 'eof':
193 raise ValueError(f"Unexpected token {self.current_token().value}")
194 return expr
195
196 def parse_or_expression(self) -> str:
197 """Parse OR and XOR expressions with same precedence and left-associative behavior.
198
199 Handles binary OR ('|', '||', 'OR', 'or') and XOR ('^', 'XOR', 'xor') operators
200 which have the same precedence level and are left-associative.
201 These are the lowest precedence operators in boolean expressions.
202
203 Returns:
204 str: The parsed OR/XOR expression in parenthesized format for BDD construction.
205 """
206 left = self.parse_and_expression()
207
208 while (self.current_token().type == 'operator' and
209 self.current_token().value in ['OR', 'XOR']):
210 op = self.current_token().value
211 self.consume_token()
212 right = self.parse_and_expression()
213 if op == 'OR':
214 left = f"({left} | {right})"
215 else: # XOR
216 left = f"({left} ^ {right})"
217
218 return left
219
220 def parse_and_expression(self) -> str:
221 """Parse AND expressions with left-associative behavior.
222
223 Handles binary AND operators ('&', '&&', 'AND', 'and') which have
224 higher precedence than OR/XOR but lower than NOT operators.
225 AND operations are left-associative.
226
227 Returns:
228 str: The parsed AND expression in parenthesized format for BDD construction.
229 """
230 left = self.parse_not_expression()
231
232 while self.current_token().type == 'operator' and self.current_token().value == 'AND':
233 self.consume_token()
234 right = self.parse_not_expression()
235 left = f"({left} & {right})"
236
237 return left
238
239 def parse_not_expression(self) -> str:
240 """Parse NOT expressions which have the highest operator precedence.
241
242 Handles unary NOT operators ('!', '~', 'NOT', 'not') which bind
243 most tightly and are right-associative (can be chained).
244 Multiple NOT operators can be applied consecutively.
245
246 Returns:
247 str: The parsed NOT expression with proper negation operators for BDD construction.
248 """
249 if self.current_token().type == 'operator' and self.current_token().value == 'NOT':
250 self.consume_token()
251 expr = self.parse_not_expression()
252 return f"~{expr}"
253 else:
254 return self.parse_primary_expression()
255
256 def parse_primary_expression(self) -> str:
257 """Parse primary expressions: variables and parenthesized sub-expressions.
258
259 Handles the most basic expression elements:
260 - Variable names: Added to the variables set for tracking
261 - Parenthesized expressions: Recursively parsed with proper grouping
262
263 Returns:
264 str: The variable name or the result of parsing the parenthesized expression.
265
266 Raises:
267 ValueError: If the current token is not a variable or opening parenthesis.
268 """
269 token = self.current_token()
270
271 if token.type == 'variable':
272 self.variables.add(token.value)
273 self.consume_token()
274 return token.value
275 elif token.type == 'lparen':
276 self.consume_token()
277 expr = self.parse_or_expression()
278 self.consume_token('rparen')
279 return expr
280 else:
281 raise ValueError(f"Expected variable or '(' but got {token.value}")

284class PythonBDDDemo:
285 """Python BDD Demo - produces output identical to C++ bdd_demo.exe.
286
287 This class provides a Python implementation that generates exactly
288 the same output formats as the C++ BDD demonstration tool, enabling
289 cross-library validation between Python 'dd' library and C++ TeDDy/CUDD.
290
291 The demo supports multiple output formats:
292 - Console output with variable mappings and statistics
293 - DOT graph files for GraphViz visualization
294 - PNG image generation (when GraphViz is available)
295 - Node analysis tables in text format
296
297 Attributes:
298 bdd (BDD): Binary Decision Diagram instance from dd library
299 variable_order (Dict[str, int]): Mapping of variables to BDD indices
300 node_table (List[Dict]): Generated node analysis data
301 """
302
303 def __init__(self):
304 self.bdd = BDD()
305 self.variable_order: Dict[str, int] = {}
306 self.node_table: List[Dict] = []

308 def read_expression_from_file(self, filename: str) -> str:
309 """Read and parse expression from file, matching C++ behavior.
310
311 Reads a boolean expression from a text file, filtering out comments
312 and empty lines. Supports multi-line expressions by joining all
313 non-comment lines.
314
315 Args:
316 filename (str): Path to file containing boolean expression
317
318 Returns:
319 str: Cleaned expression string ready for parsing
320
321 Raises:
322 FileNotFoundError: If the specified file cannot be opened
323 ValueError: If no valid expression is found in the file
324 """
325 try:
326 with open(filename, 'r', encoding='utf-8') as f:
327 lines = f.readlines()
328 except FileNotFoundError:
329 raise FileNotFoundError(f"Could not open file: {filename}")
330
331 # Filter out comments and empty lines
332 expression_lines = []
333 for line in lines:
334 line = line.strip()
335 if line and not line.startswith('#'):
336 expression_lines.append(line)
337
338 if not expression_lines:
339 raise ValueError("No expression found in file")
340
341 # Join all expression lines
342 expression = ' '.join(expression_lines)
343 return expression.strip()

345 def create_variable_ordering(self, variables: Set[str]) -> Dict[str, int]:
346 """Create consistent variable ordering for BDD construction.
347
348 Generates a deterministic alphabetical ordering of variables
349 to ensure consistent BDD structure across multiple runs.
350 This matches the C++ implementation's variable ordering strategy.
351
352 Args:
353 variables (Set[str]): Set of variable names found in expression
354
355 Returns:
356 Dict[str, int]: Mapping from variable names to BDD indices
357 """
358 sorted_vars = sorted(variables)
359 ordering = {}
360 for i, var in enumerate(sorted_vars):
361 ordering[var] = i
362 self.bdd.declare(var)
363 return ordering

365 def build_bdd_from_expression(self, expression_text: str) -> any:
366 """Build a Binary Decision Diagram from a boolean expression string.
367
368 Parses the expression text, extracts variables, creates consistent variable
369 ordering, and constructs the BDD using the 'dd' library. This method
370 coordinates the entire process from text parsing to BDD construction.
371
372 Args:
373 expression_text (str): Boolean expression in standard infix notation
374 with support for AND, OR, NOT, XOR operators and parentheses.
375
376 Returns:
377 Any: The constructed BDD object from the 'dd' library representing
378 the boolean function defined by the expression.
379
380 Side Effects:
381 - Updates self.variable_order with the variable mapping used
382 - Declares all found variables in the internal BDD manager
383 - May raise parsing errors if expression syntax is invalid
384 """
385 # Tokenize and parse
386 lexer = ExpressionLexer(expression_text)
387 parser = ExpressionParser(lexer.tokens)
388
389 # Parse to dd library format
390 dd_expression = parser.parse_expression()
391
392 # Create variable ordering
393 self.variable_order = self.create_variable_ordering(parser.variables)
394
395 # Build BDD
396 bdd_func = self.bdd.add_expr(dd_expression)
397 return bdd_func

399 def generate_node_table(self, bdd_func) -> List[Dict]:
400 """Generate node table in the same format as C++ version"""
401 nodes = []
402
403 try:
404 # Create a simple BDD representation using cofactors
405 # This is a basic implementation that creates a canonical structure
406
407 # Get the support (variables) of the BDD
408 support = self.bdd.support(bdd_func)
409
410 if not support:
411 # Constant function
412 if bdd_func == self.bdd.true:
413 nodes = [{
414 'id': 0,
415 'variable': '-',
416 'false_child': '-',
417 'true_child': '-',
418 'type': 'Terminal(1)'
419 }]
420 else:
421 nodes = [{
422 'id': 0,
423 'variable': '-',
424 'false_child': '-',
425 'true_child': '-',
426 'type': 'Terminal(0)'
427 }]
428 else:
429 # Build a simple topological representation
430 visited_functions = {}
431 node_counter = [0]
432
433 def process_function(func, depth=0):
434 """Recursively process BDD function"""
435 # Check for terminals
436 if func == self.bdd.false:
437 if 'false' not in visited_functions:
438 visited_functions['false'] = {
439 'id': node_counter[0],
440 'variable': '-',
441 'false_child': '-',
442 'true_child': '-',
443 'type': 'Terminal(0)'
444 }
445 node_counter[0] += 1
446 return visited_functions['false']['id']
447
448 if func == self.bdd.true:
449 if 'true' not in visited_functions:
450 visited_functions['true'] = {
451 'id': node_counter[0],
452 'variable': '-',
453 'false_child': '-',
454 'true_child': '-',
455 'type': 'Terminal(1)'
456 }
457 node_counter[0] += 1
458 return visited_functions['true']['id']
459
460 # Check if we've seen this function before
461 func_key = str(func)
462 if func_key in visited_functions:
463 return visited_functions[func_key]['id']
464
465 # Find the top variable
466 func_support = self.bdd.support(func)
467 if not func_support:
468 # Should not happen for non-terminal functions
469 return process_function(self.bdd.false, depth + 1)
470
471 # Choose variable with lowest index
472 top_var = min(func_support, key=lambda v: self.variable_order.get(v, 999))
473
474 # Get cofactors
475 low_func = self.bdd.let({top_var: False}, func)
476 high_func = self.bdd.let({top_var: True}, func)
477
478 # Process children first
479 low_id = process_function(low_func, depth + 1)
480 high_id = process_function(high_func, depth + 1)
481
482 # Create this node
483 node_info = {
484 'id': node_counter[0],
485 'variable': top_var,
486 'false_child': low_id,
487 'true_child': high_id,
488 'type': 'Variable'
489 }
490 visited_functions[func_key] = node_info
491 node_counter[0] += 1
492
493 return node_info['id']
494
495 # Start processing from root
496 process_function(bdd_func)
497
498 # Convert to list and reverse sort by ID to match TeDDy ordering
499 nodes = list(visited_functions.values())
500 nodes.sort(key=lambda x: x['id'], reverse=True)
501
502 # Create mapping from old IDs to new IDs
503 id_mapping = {}
504 for i, node in enumerate(nodes):
505 id_mapping[node['id']] = i
506
507 # Update all nodes with new IDs and correct child references
508 for i, node in enumerate(nodes):
509 node['id'] = i
510 if isinstance(node['false_child'], int):
511 node['false_child'] = id_mapping[node['false_child']]
512 if isinstance(node['true_child'], int):
513 node['true_child'] = id_mapping[node['true_child']]
514
515 except Exception as e:
516 print(f"Error generating node table: {e}")
517 # Create a minimal fallback based on the expression
518 support = self.bdd.support(bdd_func) if hasattr(self.bdd, 'support') else set()
519
520 if not support:
521 # Constant function fallback
522 nodes = [{
523 'id': 0,
524 'variable': '-',
525 'false_child': '-',
526 'true_child': '-',
527 'type': 'Terminal(1)'
528 }]
529 else:
530 # Create a simple tree structure based on variables
531 sorted_vars = sorted(support, key=lambda v: self.variable_order.get(v, 0))
532
533 nodes = []
534 for i, var in enumerate(sorted_vars):
535 if i == len(sorted_vars) - 1:
536 # Last variable - children are terminals
537 nodes.append({
538 'id': i,
539 'variable': var,
540 'false_child': len(sorted_vars),
541 'true_child': len(sorted_vars) + 1,
542 'type': 'Variable'
543 })
544 else:
545 # Internal variable - child is next variable
546 nodes.append({
547 'id': i,
548 'variable': var,
549 'false_child': i + 1,
550 'true_child': i + 1, # Simplified for fallback
551 'type': 'Variable'
552 })
553
554 # Add terminals
555 nodes.extend([
556 {
557 'id': len(sorted_vars),
558 'variable': '-',
559 'false_child': '-',
560 'true_child': '-',
561 'type': 'Terminal(0)'
562 },
563 {
564 'id': len(sorted_vars) + 1,
565 'variable': '-',
566 'false_child': '-',
567 'true_child': '-',
568 'type': 'Terminal(1)'
569 }
570 ])
571
572 self.node_table = nodes
573 return nodes

575 def save_node_table(self, filename: str):
576 """Save node table to file in exact C++ format"""
577 with open(filename, 'w') as f:
578 # Write header
579 f.write("Index | Variable | False Child | True Child | Type\n")
580 f.write("------|----------|-------------|------------|----------\n")
581
582 # Write nodes
583 for node in self.node_table:
584 f.write(f"{node['id']:5d} | {node['variable']:8s} | "
585 f"{str(node['false_child']):11s} | {str(node['true_child']):10s} | "
586 f"{node['type']}\n")
587
588 f.write("\n") # Empty line at end like C++ version

590 def generate_dot_graph(self, bdd_func, expression_name: str) -> str:
591 """Generate DOT graph in exact C++ format"""
592 dot_lines = ["digraph DD {", ""]
593
594 # Identify terminal nodes for shape formatting
595 terminal_nodes = []
596 for node in self.node_table:
597 if node['type'].startswith('Terminal'):
598 terminal_nodes.append(f"node{node['id']}")
599
600 if terminal_nodes:
601 dot_lines.append(f" node [shape = square] {' '.join(terminal_nodes)};")
602 dot_lines.append(" node [shape = circle];")
603 dot_lines.append("")
604
605 # Add node definitions
606 for node in self.node_table:
607 if node['type'].startswith('Terminal'):
608 value = "1" if node['type'] == 'Terminal(1)' else "0"
609 label = value
610 tooltip = value
611 else:
612 label = node['variable']
613 tooltip = str(self.variable_order.get(node['variable'], 0))
614
615 dot_lines.append(f' node{node["id"]} [label = "{label}", tooltip = "{tooltip}"];')
616
617 dot_lines.append("")
618
619 # Add edges
620 for node in self.node_table:
621 if node['type'] == 'Variable':
622 # False edge (dashed)
623 dot_lines.append(f" node{node['id']} -> node{node['false_child']} [style = dashed];")
624 # True edge (solid)
625 dot_lines.append(f" node{node['id']} -> node{node['true_child']} [style = solid];")
626
627 dot_lines.append("}")
628 dot_lines.append("") # Empty line at end
629
630 return "\n".join(dot_lines)
