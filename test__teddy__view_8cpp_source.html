11TEST_CASE("teddy_view models external_dag_view and integrates with algorithms", "[teddy][view]") {
12 using namespace teddy::ops;
13
14 // Create a small TeDDy manager with 2 variables
15 teddy::bdd_manager manager(2, 1000);
16
17 // Build a simple BDD: f = x0 AND (NOT x1) i.e. x0 & !x1
18 auto x0 = manager.variable(0);
19 auto x1 = manager.variable(1);
20 auto not_x1 = manager.apply<XOR>(x1, manager.constant(1));
21 auto f = manager.apply<AND>(x0, not_x1);
22
23 teddy_view view(&manager, f);
24
25 // Concept check
26 STATIC_REQUIRE(graph::external_dag_view<teddy_view>);
27
28 // roots should contain the root handle
29 auto roots = view.roots();
30 REQUIRE(!roots.empty());
31
32 // topo_order should return a vector of handles
33 auto order = graph::topo_order(view, roots);
34 REQUIRE(!order.empty());
35
36 // postorder_fold: count variable nodes (non-terminals)
37 auto count = graph::postorder_fold<teddy_view, int>(
38 view, order[0],
39 [](const teddy_view& g, teddy_view::handle h, std::span<const int> child_sums) {
40 int sum = 0;
41 for (auto&& e : g.children(h)) {
42 // label isn't used here
43 }
44 for (int v : child_sums)
45 sum += v;
46 // if node is non-terminal count 1, else 0
47 return h.p && !h.p->is_terminal() ? (1 + sum) : sum;
48 });
49
50 REQUIRE(count == 2);
51}
