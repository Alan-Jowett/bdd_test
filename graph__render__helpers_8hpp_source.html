45inline std::string escape_label(const std::string& s) {
46 std::string out;
47 out.reserve(s.size());
48 for (char c : s) {
49 if (c == '"')
50 out += "\\\"";
51 else
52 out += c;
53 }
54 return out;
55}

72inline std::string render_node_line(const std::string& node_id, const std::string& label,
73 const std::string& shape) {
74 std::string esc = escape_label(label);
75 if (shape == "square" || shape == "box" || shape == "rect") {
76 return std::format(" {}[\"{}\"]\n", node_id, esc);
77 } else if (shape == "circle" || shape == "ellipse" || shape == "oval") {
78 return std::format(" {}((\"{}\"))\n", node_id, esc);
79 } else if (shape == "diamond") {
80 return std::format(" {}{{\"{}\"}}\n", node_id, esc);
81 } else if (shape == "hexagon") {
82 return std::format(" {}{{{{\"{}\"}}}}\n", node_id, esc);
83 } else {
84 return std::format(" {}((\"{}\"))\n", node_id, esc);
85 }
86}

102inline std::string render_edge_line(const std::string& parent_id, const std::string& child_id,
103 const std::string& edge_style, const std::string& edge_label) {
104 std::string style = " --> ";
105 if (edge_style == "dashed")
106 style = " -.-> ";
107 else if (edge_style == "dotted")
108 style = " -..-> ";
109 else if (edge_style == "thick")
110 style = " ==> ";
111
112 if (!edge_label.empty()) {
113 return std::format(" {}{} |\"{}\"| {}\n", parent_id, style, escape_label(edge_label),
114 child_id);
115 }
116 return std::format(" {}{}{}\n", parent_id, style, child_id);
117}

130inline std::vector<std::pair<std::string, std::string>> flatten_and_sort_class_assignments(
131 const std::unordered_map<std::string, std::vector<std::string>>& class_to_nodes) {
132 std::vector<std::pair<std::string, std::string>> out;
133 out.reserve(256);
134 for (const auto& p : class_to_nodes) {
135 const auto& class_name = p.first;
136 const auto& ids = p.second;
137 std::transform(ids.begin(), ids.end(), std::back_inserter(out),
138 [&](const std::string& id) { return std::make_pair(id, class_name); });
139 }
140 std::sort(out.begin(), out.end(), [](const auto& a, const auto& b) {
141 return graph_common::node_id_less(a.first, b.first);
142 });
143 return out;
144}

153inline std::string join_dot_properties(const std::vector<std::string>& props) {
154 if (props.empty())
155 return std::string();
156 std::string out = "[";
157 for (size_t i = 0; i < props.size(); ++i) {
